\begin{theorem}[Conjugacy classes of order-$q$ subgroups of $\operatorname{GL}_2(\mathbb{F}_p)$]\label{thm:conj-classes-order-q}
\lean{ConjugacyClassesOrderQ.conjClasses_order_q_GL_two}
\leanfile{M2rGroup7/ConjugacyClassesOrderQ.lean}
Let $p$ and $q$ be distinct primes with $q > 2$. The number of conjugacy classes of subgroups of order $q$ in $G = \operatorname{GL}_2(\mathbb{F}_p)$ is
\[
  s_q\!\bigl(\operatorname{GL}_2(\mathbb{F}_p)\bigr)
  \;=\;
  \begin{cases}
    \dfrac{q+3}{2}, & \text{if } q \mid (p-1),\\[4pt]
    1, & \text{if } q \mid (p+1),\\[4pt]
    0, & \text{otherwise.}
  \end{cases}
\]
Equivalently, with $\mathbf{1}_{q\mid n}$ the indicator,
\[
  s_q\!\bigl(\operatorname{GL}_2(\mathbb{F}_p)\bigr)
  \;=\;
  \tfrac{q+3}{2}\,\mathbf{1}_{q\mid (p-1)} \;+\; \mathbf{1}_{q\mid (p+1)} .
\]
\end{theorem}

\begin{proof}
Set $G = \operatorname{GL}_2(\mathbb{F}_p)$ and let $V = \mathbb{F}_p^2$. Any subgroup $H \le G$ of order $q$ is cyclic since $q$ is prime, so $H = \langle M \rangle$ for some $M \in G$ of order $q$. Since $q > 2$ and $p,q$ are distinct primes, $q$ cannot divide both $p-1$ and $p+1$ (else $q \mid 2$). This gives two mutually exclusive cases.

\medskip\noindent\emph{Case 1: $q \mid (p+1)$, so $q \nmid (p-1)$.} By Lemma~\ref{lem:case-pp1}, all subgroups of order $q$ are conjugate, giving exactly $1$ class.

\medskip\noindent\emph{Case 2: $q \mid (p-1)$, so $q \nmid (p+1)$.} By Lemma~\ref{lem:case-pm1}, the number of conjugacy classes is $\tfrac{q+3}{2}$.

\medskip\noindent\emph{Other case: $q \nmid (p-1)$ and $q \nmid (p+1)$.} By Lemma~\ref{lem:no-order-q}, there are no elements of order $q$ in $G$, hence no subgroups of order $q$ and $0$ classes.

Combining yields the stated formula. $\blacksquare$
\end{proof}

\begin{lemma}\label{lem:order-divides-p2-1}
\lean{ConjugacyClassesOrderQ.order_q_divides_p2_sub_one}
\leanfile{M2rGroup7/ConjugacyClassesOrderQ.lean}
\leanok
Let $p,q$ be distinct primes. If $M \in \operatorname{GL}_2(\mathbb{F}_p)$ has order $q$, then $q \mid (p^2 - 1) = (p-1)(p+1)$.
\end{lemma}

\begin{proof}
The order of $\operatorname{GL}_2(\mathbb{F}_p)$ is $(p^2-1)(p^2-p) = p\,(p-1)^2(p+1)$. By Lagrange, $q \mid |G|$. Since $q \ne p$ is prime, $q \mid (p-1)^2(p+1)$, hence $q \mid (p-1)(p+1) = p^2-1$.
\end{proof}

\begin{lemma}\label{lem:no-order-q}
\lean{ConjugacyClassesOrderQ.no_subgroup_of_order_q}
\leanfile{M2rGroup7/ConjugacyClassesOrderQ.lean}
\leanok
Let $p,q$ be distinct primes with $q > 2$, $q \nmid (p-1)$ and $q \nmid (p+1)$. Then $\operatorname{GL}_2(\mathbb{F}_p)$ has no element of order $q$, hence no subgroup of order $q$.
\end{lemma}

\begin{proof}
By Lemma~\ref{lem:order-divides-p2-1}, an element of order $q$ would force $q \mid (p-1)(p+1)$, which contradicts the hypothesis.
\end{proof}

\begin{lemma}\label{lem:case-pp1}
\lean{ConjugacyClassesOrderQ.case_q_dvd_p_plus_one}
\leanfile{M2rGroup7/ConjugacyClassesOrderQ.lean}
\leanok
Let $p,q$ be distinct primes with $q > 2$ and $q \mid (p+1)$ (so $q \nmid (p-1)$). Then any two subgroups of $\operatorname{GL}_2(\mathbb{F}_p)$ of order $q$ are conjugate, and there exists at least one such subgroup.
\end{lemma}

\begin{proof}
We first show that any $M \in G$ of order $q$ has no eigenvalues in $\mathbb{F}_p$. If $\lambda \in \mathbb{F}_p$ is an eigenvalue, then $\lambda^q = 1$ in $\mathbb{F}_p^\times$, forcing the order of $\lambda$ to divide $q$. Since $q \nmid (p-1)$ the order is $1$, so $\lambda = 1$. Both eigenvalues of $M$ in $\overline{\mathbb{F}_p}$ would then be $1$. Since $\gcd(q,p)=1$, $x^q-1$ is separable, so $M$ is diagonalizable and equals $I$, contradicting order $q$.

Thus the characteristic polynomial of $M$ is irreducible of degree $2$, and over $K = \mathbb{F}_p[x]/(f) \cong \mathbb{F}_{p^2}$ its roots are a primitive $q$-th root of unity $\omega$ and its Frobenius conjugate $\omega^p$. Since $K^\times$ is cyclic of order $p^2-1$ and $q \mid p^2-1$, the group of $q$-th roots of unity in $K$ is cyclic of order $q$, generated by $\omega$.

For any other $M' \in G$ of order $q$ with eigenvalues $\omega^b, \omega^{bp}$, choose $k$ with $kb \equiv a \pmod q$. Then $M'^k$ has eigenvalues $\omega^a, \omega^{ap}$, i.e.\ the same characteristic polynomial as $M$. Picking any $v \ne 0$, $\{v, Mv\}$ is a basis (since $M$ has no eigenline), and in this basis $M$ acts as a fixed companion matrix $C$. So $M$ and $M'^k$ are conjugate to $C$, hence to each other, and the subgroups they generate are conjugate.

For existence: $\omega + \omega^p \in \mathbb{F}_p$ (it is Frobenius-fixed) and $\omega\,\omega^p = \omega^{p+1} = 1$ since $q \mid p+1$. So the polynomial $x^2 - tx + 1$ with $t = \omega + \omega^p$ is irreducible over $\mathbb{F}_p$, and its companion matrix lies in $G$ and has order $q$.
\end{proof}

\begin{lemma}\label{lem:case-pm1}
\lean{ConjugacyClassesOrderQ.case_q_dvd_p_minus_one}
\leanfile{M2rGroup7/ConjugacyClassesOrderQ.lean}
Let $p,q$ be distinct primes with $q > 2$ and $q \mid (p-1)$ (so $q \nmid (p+1)$). Then the number of conjugacy classes in $\operatorname{GL}_2(\mathbb{F}_p)$ of subgroups of order $q$ equals $\tfrac{q+3}{2}$.
\end{lemma}

\begin{proof}
By Lemma~\ref{lem:case-pp1}'s contrapositive: if $M$ has order $q$ and no eigenvalue in $\mathbb{F}_p$, then $q \mid p+1$, contradiction. Hence $M$ has an eigenvalue $\lambda \in \mathbb{F}_p^\times$ of order $q$, and since $\gcd(q,p)=1$ and $M^q=I$, $M$ is diagonalizable over $\mathbb{F}_p$. Up to conjugacy we may assume $H$ is contained in the diagonal torus $D \cong C_{p-1}^2$.

Let $a \in \mathbb{F}_p^\times$ be a fixed element of order $q$. The subgroups of order $q$ in $D$ are exactly
\[
  U_\ell := \langle \operatorname{diag}(a, a^\ell) \rangle, \quad \ell \in \mathbb{F}_q,
  \qquad
  U_\infty := \langle \operatorname{diag}(1, a) \rangle.
\]
The normalizer of $D$ in $G$ is the monomial group $N(D) = D \rtimes \langle \sigma \rangle$ where $\sigma$ swaps coordinates. Conjugation by $\sigma$ sends $U_\ell \mapsto U_{\ell^{-1}}$ for $\ell \in \mathbb{F}_q^\times$ and exchanges $U_0 \leftrightarrow U_\infty$.

Moreover, by the Bruhat / double-coset description of $G$, any two diagonal subgroups of order $q$ conjugate in $G$ are already conjugate in $N(D)$. Hence the set of conjugacy classes is the set of orbits of the involution $\ell \mapsto \ell^{-1}$ on $\mathbb{F}_q \cup \{\infty\}$, where $0 \leftrightarrow \infty$.

The orbits are: $\{0,\infty\}$, $\{1\}$, $\{-1\}$ (using $q>2$), and $\tfrac{q-3}{2}$ pairs $\{\ell,\ell^{-1}\}$ for $\ell \in \mathbb{F}_q^\times \setminus \{\pm 1\}$. Total
\[
  1 + 1 + 1 + \frac{q-3}{2} \;=\; \frac{q+3}{2}.
\]
\end{proof}
