\section{Setup}

We follow Wild's paper ``The Groups of Order Sixteen Made Easy'' (Marcel Wild,
\emph{Amer.\ Math.\ Monthly} \textbf{112} (2005), 20--31). Throughout, $G$ is a
finite group of order $16$ and is not isomorphic to $(C_2)^4$. The goal is to
show that $G$ realises one of the $13$ cyclic-extension types
$\mathtt{ext\_16\_1}, \dots, \mathtt{ext\_16\_13}$ already defined in
\texttt{M2rGroup7/SixteenClassification/Blueprints.lean}.

We use the following data structures from
\texttt{M2rGroup7/SixteenClassification/Extensions.lean}:

\begin{itemize}
\item An \emph{extension type} $E = (N, n, \tau, v)$ packages a group $N$, an
exponent $n \in \mathbb{N}$, an automorphism $\tau \in \mathrm{Aut}(N)$ fixing
$v \in N$, with $\tau^n = \mathrm{conj}_v$.
\item A \emph{realisation} of $E$ by a group $G$ is a tuple
$(a, \iota)$ where $a \in G$, $\iota : N \hookrightarrow G$ is a monoid
homomorphism, $a \iota(x) a^{-1} = \iota(\tau(x))$ for all $x \in N$,
$a^n = \iota(v)$, and the map $(x, i) \mapsto \iota(x) \cdot a^i$ is a bijection
$N \times \{0, 1, \dots, n-1\} \xrightarrow{\sim} G$.
\end{itemize}

The thirteen extension types $\mathtt{ext\_16\_i}$ are pre-defined; each one is
realised by an explicit order-$16$ group in
\texttt{M2rGroup7/SixteenClassification/Blueprints.lean}.

\section{Transfer along isomorphisms}

\begin{definition}
  \label{def:realise-transfer}
  \lean{OrderSixteen.RealiseExtType.transfer}
  \leanfile{M2rGroup7/SixteenClassification/Lemma3.lean}
  \leanok
  Let $G$ and $G'$ be groups, $E$ an extension type, $e : G \simeq G'$ a group
  isomorphism, and $R : \mathrm{RealiseExtType}~G'~E$. Then the
  \emph{transferred realisation} $\mathrm{transfer}(e, R)$ is the canonical
  realisation of $E$ by $G$ obtained by pulling $R$ back along $e$.
  Concretely, take $a := e^{-1}(R.a)$, $\iota := e^{-1} \circ R.\iota$,
  and the bijection $(x, i) \mapsto \iota(x) \cdot a^i$ as $R.\mathrm{equiv}$
  composed with $e^{-1}$.
\end{definition}

\begin{lemma}
  \label{lem:realise-of-mulEquiv}
  \lean{OrderSixteen.realise_of_mulEquiv}
  \leanfile{M2rGroup7/SixteenClassification/Lemma3.lean}
  \uses{def:realise-transfer}
  \leanok
  If $G \simeq G'$ and $G'$ admits a realisation of an extension type $E$, then
  so does $G$.
\end{lemma}

\begin{proof}
  Immediate from Definition~\ref{def:realise-transfer}.
\end{proof}

\section{Inducing elements}

\begin{lemma}
  \label{lem:inducing-element-exists}
  \lean{OrderSixteen.exists_inducing_element}
  \leanfile{M2rGroup7/SixteenClassification/Lemma3.lean}
  Let $G$ be a group and $H \triangleleft G$ a normal subgroup of index $2$.
  Then there exists $a \in G \setminus H$, and any such $a$ satisfies
  $a^2 \in H$ and $\mathrm{conj}_a(H) = H$. The map $h \mapsto a h a^{-1}$ is
  therefore an automorphism of $H$.
\end{lemma}

\begin{proof}
  Since $[G : H] = 2 > 1$, $H \neq G$, so some $a \notin H$ exists. The coset
  $Ha$ has order dividing $[G : H] = 2$ and is non-identity, hence order $2$.
  Therefore $(Ha)^2 = H$, i.e.\ $a^2 \in H$. Normality of $H$ gives
  $aHa^{-1} = H$ and the conjugation map $h \mapsto aha^{-1}$ restricts to an
  automorphism of $H$.
\end{proof}

\begin{lemma}
  \label{lem:realise-from-normal-index-2}
  \lean{OrderSixteen.realise_from_normal_index_two}
  \leanfile{M2rGroup7/SixteenClassification/Lemma3.lean}
  \uses{lem:inducing-element-exists}
  Let $G$ be a finite group, $H \triangleleft G$ a normal subgroup of index $2$,
  and $a \in G \setminus H$. Set $v := a^2 \in H$ and let
  $\tau \in \mathrm{Aut}(H)$ be conjugation by $a$. Then the extension type
  $E := (H, 2, \tau, v)$ is well-defined ($\tau(v) = v$ and $\tau^2 =
  \mathrm{conj}_v$) and $G$ realises $E$ with inducing element $a$ and
  embedding $\iota : H \hookrightarrow G$.
\end{lemma}

\begin{proof}
  Set $v := a^2$ and $\tau(h) := aha^{-1}$. Then $\tau(v) = a v a^{-1} = a^3
  a^{-1} = a^2 = v$ and $\tau^2(h) = a^2 h a^{-2} = v h v^{-1} = \mathrm{conj}_v
  (h)$. The decomposition $G = H \sqcup Ha$ as cosets of $H$ gives the
  bijection $H \times \{0, 1\} \xrightarrow{\sim} G$,
  $(h, i) \mapsto h \cdot a^i$, completing the data of a realisation.
\end{proof}

\section{Case analysis for normal \texorpdfstring{$C_8$}{C8}}

\begin{lemma}
  \label{lem:realise-with-normal-C8}
  \lean{OrderSixteen.realise_with_normal_C8}
  \leanfile{M2rGroup7/SixteenClassification/Lemma3.lean}
  \uses{lem:realise-from-normal-index-2, lem:realise-of-mulEquiv}
  Let $G$ be a group of order $16$ with a normal subgroup $H \le G$ such that
  $H \simeq C_8$. Then $G$ realises one of the extension types
  $\mathtt{ext\_16\_1}$, $\mathtt{ext\_16\_5}$, $\mathtt{ext\_16\_6}$,
  $\mathtt{ext\_16\_7}$, $\mathtt{ext\_16\_8}$, or $\mathtt{ext\_16\_9}$.
\end{lemma}

\begin{proof}
  Write $H = \langle x \rangle$ with $|x| = 8$. The hypotheses $|G| = 16$ and
  $|H| = 8$ give $[G : H] = 2$, so $H$ is normal of index $2$ and Lagrange
  gives a partition $G = H \sqcup Ha$ for any chosen
  $a \in G \setminus H$.

  By Lemma~\ref{lem:inducing-element-exists} the set $G \setminus H$ is
  non-empty, so the set
  $\{\,|b| \mid b \in G \setminus H\,\}$ is a non-empty subset of the positive
  divisors of $|G| = 16$, hence has a minimum. Choose an inducing element
  $a \in G \setminus H$ of \emph{minimal order}. Because $a \notin H$ but
  $a^2 \in H$ (since the coset $Ha$ has order $2$ in $G/H$), the order of $a$
  is even, and as $|a|$ divides $|G| = 16$ we have
  $|a| \in \{2, 4, 8, 16\}$.

  Set $v := a^2 \in H$ and let $\tau := \mathrm{conj}_a \in \mathrm{Aut}(H)$ be
  conjugation by $a$ restricted to $H$. By
  Lemma~\ref{lem:realise-from-normal-index-2}, $G$ realises the extension type
  $(H, 2, \tau, v)$.

  Recall (Wild's Fact~3 for $\mathrm{Aut}(C_8)$) that
  $\mathrm{Aut}(H) \cong (\mathbb{Z}/8\mathbb{Z})^\times =
  \{1, 3, 5, 7\}$ as a multiplicative group, with $\tau$ acting as $x \mapsto
  x^k$ for one of the four odd residues $k$. Every such $k$ satisfies
  $k^2 \equiv 1 \pmod 8$, so every $\tau \in \mathrm{Aut}(H)$ is an involution.

  We split on $|a|$.

  \smallskip
  \noindent
  \emph{Case 1: $|a| = 2$.} Then $v = a^2 = e$, so $\mathrm{conj}_v$ is the
  identity automorphism of $H$, and the validity condition $\tau^2 =
  \mathrm{conj}_v$ becomes $\tau^2 = \mathrm{id}$. This is automatic since every
  $\tau \in \mathrm{Aut}(H)$ is an involution. The four choices
  $\tau \in \{1, \cdot^3, \cdot^5, \cdot^7\}$ give four extension types
  $(H, 2, \tau, e)$, matching exactly the four predefined types
  \[
    \mathtt{ext\_16\_5} = (C_8, 2, \mathrm{id}, e),\quad
    \mathtt{ext\_16\_8} = (C_8, 2, \cdot^3, e),\quad
    \mathtt{ext\_16\_6} = (C_8, 2, \cdot^5, e),\quad
    \mathtt{ext\_16\_7} = (C_8, 2, \cdot^7, e),
  \]
  realised respectively by $C_8 \times C_2$, $C_8 \rtimes_{\mathrm{pow}_3} C_2$,
  $C_8 \rtimes_{\mathrm{pow}_5} C_2$, and the dihedral group $D_8$. In each
  sub-case the realisation of $G$ is obtained directly from the inducing data
  $(a, \iota)$ produced by
  Lemma~\ref{lem:realise-from-normal-index-2}.

  \smallskip
  \noindent
  \emph{Case 2: $|a| = 4$.} Then $v = a^2$ has order $2$, and the only
  involution in $H = \langle x \rangle \simeq C_8$ is $x^4$, so $v = x^4$.
  Minimality means no element of $G \setminus H$ has order $2$, otherwise we
  would be in Case~1. The validity constraint $\tau(v) = v$ is automatic:
  $\tau(x^4) = x^{4k} = x^4$ for all odd $k$ (since $4k \equiv 4 \pmod 8$).
  The validity constraint $\tau^2 = \mathrm{conj}_{x^4} = \mathrm{id}$ (as
  $H$ is abelian) is also automatic. So all four
  $\tau \in \{1, \cdot^3, \cdot^5, \cdot^7\}$ pass the validity checks, giving
  candidate extension types $(C_8, 2, \tau, x^4)$. We show that the minimality
  assumption rules out three of them:
  \begin{itemize}
    \item $\tau = \mathrm{id}$: then $a x = x a$, and $b := x^{-2} a$
    satisfies $b^2 = x^{-4} a^2 = x^{-4} x^4 = e$, so $b$ is an involution. As
    $a \notin H$ and $x^{-2} \in H$, $b \notin H$, contradicting minimality.
    \item $\tau = \cdot^3$: then $a x a^{-1} = x^3$, and $(xa)^2 = x \cdot
    a x a^{-1} \cdot a^2 = x \cdot x^3 \cdot x^4 = x^8 = e$. Thus $xa$ is an
    involution in $G \setminus H$, contradicting minimality.
    \item $\tau = \cdot^5$: then $a x^2 a^{-1} = (a x a^{-1})^2 = x^{10}$, and
    $(x^2 a)^2 = x^2 \cdot a x^2 a^{-1} \cdot a^2 = x^2 \cdot x^{10} \cdot x^4
    = x^{16} = e$. Thus $x^2 a$ is an involution in $G \setminus H$,
    contradicting minimality.
    \item $\tau = \cdot^7$ (i.e.\ inversion): for every $k$,
    $(x^k a)^2 = x^k \cdot a x^k a^{-1} \cdot a^2 = x^k \cdot x^{-k} \cdot x^4
    = x^4 \neq e$, and $(x^k a)^4 = (x^4)^2 = x^8 = e$, so every element of
    $G \setminus H$ has order $4$. This satisfies the minimality assumption.
  \end{itemize}
  Hence only $\tau = \cdot^7$ survives in Case~2, yielding the extension type
  $\mathtt{ext\_16\_9} = (C_8, 2, \cdot^7, x^4)$. The corresponding realising
  group is the generalised quaternion group $Q_{16}$.

  \smallskip
  \noindent
  \emph{Case 3: $|a| = 8$.} Then $v = a^2$ has order $4$, so
  $v \in \{x^2, x^6\}$. The two values are related by $x^6 = (x^2)^3$, and
  the automorphism $\cdot^3$ sends $x^2$ to $x^6$. The action set
  $\{\mathrm{id}, \cdot^3, \cdot^5, \cdot^7\}$ is a union of
  $\mathrm{Aut}(H)$-conjugacy classes (every element is in its own class
  because $\mathrm{Aut}(H)$ is abelian, but this is enough). Hence by
  Lemma~\ref{lem:realise-of-mulEquiv} together with
  \texttt{extension\_families\_same\_isoClasses} we may assume $v = x^2$.

  The validity constraint $\tau(v) = v$ now reads $x^{2k} = x^2$, i.e.\
  $2(k - 1) \equiv 0 \pmod 8$, i.e.\ $k \equiv 1 \pmod 4$, so
  $k \in \{1, 5\}$ and $\tau \in \{\mathrm{id}, \cdot^5\}$.
  Minimality forbids elements of order $\le 4$ in $G \setminus H$, otherwise
  we would be in Case~1 or Case~2. We rule out each remaining $\tau$:
  \begin{itemize}
    \item $\tau = \mathrm{id}$: then $a$ commutes with $H$, and $b := x^{-1}
    a$ satisfies $b^2 = x^{-2} a^2 = x^{-2} \cdot x^2 = e$. Thus $b$ is an
    involution in $G \setminus H$, contradicting minimality.
    \item $\tau = \cdot^5$: then $a x a^{-1} = x^5$, and $(xa)^2 = x \cdot
    a x a^{-1} \cdot a^2 = x \cdot x^5 \cdot x^2 = x^8 = e$. Thus $xa$ is an
    involution in $G \setminus H$, contradicting minimality.
  \end{itemize}
  No surviving extension type arises in Case~3: any group $G$ falling into
  this case is already covered by Cases~1, 2, or 4 via the corresponding
  smaller-order inducing element.

  \smallskip
  \noindent
  \emph{Case 4: $|a| = 16$.} Then $\langle a \rangle = G$, so $G \simeq
  C_{16}$. Note that $a^2$ generates $\langle a \rangle^2 = $ the unique
  subgroup of $G$ of order $8$, which must be $H$. Thus, writing $x = a^2$,
  the extension type associated to $a$ is
  $(C_8, 2, \mathrm{id}, x) = \mathtt{ext\_16\_1}$ (the action is trivial
  because $G$ is abelian). The group $C_{16}$ comes with the realisation
  $\mathtt{realise\_16\_1}$ from
  \texttt{M2rGroup7/SixteenClassification/Blueprints.lean}, and
  Lemma~\ref{lem:realise-of-mulEquiv} applied to a chosen isomorphism
  $G \simeq C_{16}$ transfers this to a realisation of $\mathtt{ext\_16\_1}$
  by $G$.

  \smallskip
  Combining the four cases, $G$ realises one of $\mathtt{ext\_16\_1}$,
  $\mathtt{ext\_16\_5}$, $\mathtt{ext\_16\_6}$, $\mathtt{ext\_16\_7}$,
  $\mathtt{ext\_16\_8}$, or $\mathtt{ext\_16\_9}$, as required.
\end{proof}

\section{Case analysis for normal \texorpdfstring{$K_8 = C_4 \times C_2$}{K8 = C4 x C2}}

\begin{lemma}
  \label{lem:realise-with-normal-K8}
  \lean{OrderSixteen.realise_with_normal_K8}
  \leanfile{M2rGroup7/SixteenClassification/Lemma3.lean}
  \uses{lem:realise-from-normal-index-2, lem:realise-of-mulEquiv}
  Let $G$ be a group of order $16$ with a normal subgroup $H \le G$ such that
  $H \simeq K_8 := C_4 \times C_2$. Then $G$ realises one of the extension
  types $\mathtt{ext\_16\_2}$, $\mathtt{ext\_16\_3}$, $\mathtt{ext\_16\_4}$,
  $\mathtt{ext\_16\_10}$, $\mathtt{ext\_16\_11}$, $\mathtt{ext\_16\_12}$, or
  $\mathtt{ext\_16\_13}$.
\end{lemma}

\begin{proof}
  Write $H = \langle x, y \rangle$ with $|x| = 4$, $|y| = 2$, and $xy = yx$,
  so that $H = \langle x \rangle \times \langle y \rangle \simeq C_4 \times
  C_2$. The hypothesis $|G| = 16$ together with $|H| = 8$ gives $[G : H] = 2$,
  so $H$ is normal of index $2$ and Lagrange yields a partition $G = H \sqcup
  Ha$ for any $a \in G \setminus H$.

  By Lemma~\ref{lem:inducing-element-exists}, $G \setminus H$ is non-empty and
  every $b \in G \setminus H$ satisfies $b^2 \in H$, so $|b|$ is even and
  divides $16$. Hence $|b| \in \{2, 4, 8, 16\}$.

  \smallskip
  \noindent
  \emph{Reduction to $|a| \in \{2, 4\}$.} We may assume $G$ has no element of
  order $8$ (and a fortiori no element of order $16$). Indeed, if some
  $g \in G$ has order $8$ then $\langle g \rangle$ is a subgroup of $G$ of
  index $2$, hence normal, and isomorphic to $C_8$. In that situation the
  theorem
  \texttt{realise\_ext\_type\_if\_not\_iso\_to\_C2\_4}~(Theorem~\ref{thm:realise-ext-type-if-not-iso-to-C2-4})
  dispatches via Lemma~\ref{lem:realise-with-normal-C8} rather than the
  present lemma. Choosing $a$ to be an inducing element of minimal order in
  $G \setminus H$ therefore gives $|a| \in \{2, 4\}$.

  Set $v := a^2 \in H$ and let $\tau := \mathrm{conj}_a \in \mathrm{Aut}(H)$.
  By Lemma~\ref{lem:realise-from-normal-index-2}, $G$ realises the extension
  type $(H, 2, \tau, v)$.

  \smallskip
  \noindent
  \emph{Structure of $\mathrm{Aut}(H)$.} $\mathrm{Aut}(K_8) \simeq D_4$ has
  order $8$. The Lean blueprint uses three named involutions:
  \[
    \psi_3 : (x, y) \mapsto (x^{-1}, y), \qquad
    \psi_5 : (x, y) \mapsto (xy, y), \qquad
    \psi_6 : (x, y) \mapsto (x^{-1}, x^2 y).
  \]
  A short computation shows $\psi_3$ is central in $\mathrm{Aut}(H)$ (it
  commutes with both $\psi_5$ and $\psi_6$), and the kernel of the natural map
  $\mathrm{Aut}(H) \to \mathrm{Aut}(H[2]) \simeq S_3$ (where $H[2] = \{e, x^2,
  y, x^2 y\}$ is the unique Klein four-subgroup) is
  $\{\mathrm{id}, \psi_3, \psi_5, \psi_3 \psi_5\}$. Consequently the
  conjugacy classes of involutions in $\mathrm{Aut}(H)$ are
  \[
    \{\mathrm{id}\}, \quad \{\psi_3\}, \quad \{\psi_5, \psi_3 \psi_5\},
    \quad \{\psi_6, \psi_3 \psi_6\},
  \]
  giving four conjugacy classes of order-$\le 2$ elements. The element $x^2$
  is characteristic in $H$ (it is the unique square of an order-$4$ element),
  so it is fixed by every $\alpha \in \mathrm{Aut}(H)$; the orbit of $y$ under
  $\mathrm{Aut}(H)$ is $\{y, x^2 y\}$ (with $\psi_6$ interchanging them).

  \smallskip
  We split on $|a|$.

  \smallskip
  \noindent
  \emph{Case 1: $|a| = 2$.} Then $v = a^2 = e$, so the validity condition
  $\tau^2 = \mathrm{conj}_v = \mathrm{id}$ just says that $\tau$ is an
  involution (or the identity) in $\mathrm{Aut}(H)$. By the conjugacy-class
  description above, the four
  $\mathrm{Aut}(H)$-conjugacy classes of involutions $\cup \{\mathrm{id}\}$
  give, by Lemma~1(c) (formalised as \texttt{conjugateActEquiv}/
  \texttt{extension\_families\_same\_isoClasses}), four equivalence classes
  of extension types $(H, 2, \tau, e)$ with representatives
  \[
    \tau = \mathrm{id},\quad \tau = \psi_3,\quad \tau = \psi_5,\quad \tau =
    \psi_6,
  \]
  matching exactly the four predefined types
  \[
    \mathtt{ext\_16\_10} = (K_8, 2, \mathrm{id}, e),\quad
    \mathtt{ext\_16\_11} = (K_8, 2, \psi_3, e),
  \]
  \[
    \mathtt{ext\_16\_3} = (K_8, 2, \psi_5, e),\quad
    \mathtt{ext\_16\_13} = (K_8, 2, \psi_6, e),
  \]
  realised respectively by $K_8 \times C_2$, $C_2 \times D_4$,
  $(C_2 \times C_2) \rtimes_{c_4} C_4$, and the semidirect product
  $K_8 \rtimes_{\psi_6} C_2$.

  \smallskip
  \noindent
  \emph{Case 2: $|a| = 4$.} Then $v = a^2$ has order $2$ in $H$, so
  $v \in \{x^2, y, x^2 y\}$. Because $\psi_6$ interchanges $y$ and $x^2 y$,
  Lemma~\ref{lem:realise-of-mulEquiv} together with
  \texttt{conjugateActEquiv} reduces the sub-cases $v = y$ and $v = x^2 y$ to
  each other. We may therefore assume $v \in \{x^2, y\}$.

  \smallskip
  \noindent
  \emph{Sub-case 2a: $v = x^2$.} Since $x^2$ is characteristic, every
  $\alpha \in \mathrm{Aut}(H)$ stabilises $v$, and $\tau^2 = \mathrm{conj}_{x^2}
  = \mathrm{id}$ (as $H$ is abelian) forces $\tau$ to be the identity or an
  involution. By Lemma~1(c) we may take $\tau$ to be a representative of one
  of the four conjugacy classes $\{\mathrm{id}\}, \{\psi_3\}, \{\psi_5\},
  \{\psi_6\}$. For each $b = x^k y^l a \in G \setminus H$ a direct computation
  gives
  \[
    (x^k y^l a)^2 = x^k y^l \cdot \tau(x^k y^l) \cdot v.
  \]
  \begin{itemize}
    \item $\tau = \mathrm{id}$: $\tau(x^k y^l) = x^k y^l$, so $(b)^2 = x^{2k}
    y^{2l} \cdot x^2 = x^{2k + 2}$. Taking $k = -1, l = 0$ gives $(x^{-1} a)^2
    = e$, so $x^{-1} a$ is an involution in $G \setminus H$, contradicting
    minimality.
    \item $\tau = \psi_3$: $\tau(x^k y^l) = x^{-k} y^l$, so $b \cdot \tau(b) =
    y^{2l} = e$ and hence $(b)^2 = x^2$ for every $h = x^k y^l$. Thus every
    element of $G \setminus H$ has order $4$ (since $(b)^4 = x^4 = e$),
    confirming minimality. The resulting extension type is
    $\mathtt{ext\_16\_12} = (K_8, 2, \psi_3, x^2)$, realised by
    $C_2 \times Q_2$.
    \item $\tau = \psi_5$: $\tau(x^k y^l) = x^k y^{k+l}$, so $b \cdot \tau(b)
    = x^{2k} y^{2l + k} = x^{2k} y^{k \bmod 2}$, giving
    $(b)^2 = x^{2k + 2} y^{k \bmod 2}$. This equals $e$ iff $2k + 2 \equiv 0
    \pmod 4$ (i.e.\ $k$ odd) and $k \equiv 0 \pmod 2$ (i.e.\ $k$ even),
    which is impossible. Hence every element of $G \setminus H$ has order
    exactly $4$ and minimality is preserved. The resulting extension type is
    $\mathtt{ext\_16\_4} = (K_8, 2, \psi_5, x^2)$, realised by $C_4 \rtimes
    _{\mathrm{inv}} C_4$.
    \item $\tau = \psi_6$: $\tau(x^k y^l) = x^{-k} (x^2 y)^l = x^{2l - k}
    y^l$, so $b \cdot \tau(b) = x^{2l} y^{2l} = x^{2l}$, giving $(b)^2 =
    x^{2l + 2}$. Taking $l = 1, k = 0$ gives $(y a)^2 = e$, so $y a$ is an
    involution in $G \setminus H$, contradicting minimality.
  \end{itemize}
  Hence Sub-case~2a yields the two surviving extension types
  $\mathtt{ext\_16\_4}$ and $\mathtt{ext\_16\_12}$.

  \smallskip
  \noindent
  \emph{Sub-case 2b: $v = y$.} The stabiliser of $y$ in $\mathrm{Aut}(H)$ is
  the kernel of the map $\mathrm{Aut}(H) \to \mathrm{Aut}(H[2])$ described
  above, namely $V = \{\mathrm{id}, \psi_3, \psi_5, \psi_3 \psi_5\}$. Hence
  validity $\tau(v) = v$ together with $\tau^2 = \mathrm{conj}_y = \mathrm{id}$
  restricts $\tau$ to $V$. Lemma~1(c) only allows conjugation by elements of
  $V$, but $V$ is abelian, so each of the four $\tau \in V$ must be analysed
  separately. As in Sub-case~2a we use
  \[
    (x^k y^l a)^2 = x^k y^l \cdot \tau(x^k y^l) \cdot v.
  \]
  \begin{itemize}
    \item $\tau = \mathrm{id}$: $b \cdot \tau(b) = x^{2k} y^{2l} = x^{2k}$, so
    $(b)^2 = x^{2k} y$. For $k$ even this is $y$, and for $k$ odd this is
    $x^2 y$. In all cases $(b)^2 \neq e$, so every element of $G \setminus H$
    has order $4$ and minimality is preserved. The resulting extension type
    is $\mathtt{ext\_16\_2} = (K_8, 2, \mathrm{id}, y)$, realised by $C_4
    \times C_4$.
    \item $\tau = \psi_5$: by Lemma~1(b) (the cocycle-shift
    \texttt{extension\_families\_same\_isoClasses}), $(\psi_5, y) \sim (\psi_5,
    y \cdot h \cdot \psi_5(h))$ for any $h \in H$. Taking $h = x$ gives
    $y \cdot x \cdot \psi_5(x) = y \cdot x \cdot xy = x^2$, so $(\psi_5, y) \sim
    (\psi_5, x^2)$, which is $\mathtt{ext\_16\_4}$ and is already covered by
    Sub-case~2a.
    \item $\tau = \psi_3 \psi_5$: write $\sigma = \psi_3 \psi_5$, so
    $\sigma(x^k y^l) = x^{-k} y^{k + l}$ and $b \cdot \sigma(b) = y^{2l + k} =
    y^{k \bmod 2}$, giving $(b)^2 = y \cdot y^{k \bmod 2}$. Taking $k = 1$
    gives $(x a)^2 = e$, so $x a$ is an involution in $G \setminus H$,
    contradicting minimality.
    \item $\tau = \psi_3$: $b \cdot \tau(b) = y^{2l} = e$, so $(b)^2 = y$ and
    every element of $G \setminus H$ has order $4$. Minimality is preserved,
    but the resulting extension type $(K_8, 2, \psi_3, y)$ is not one of the
    seven $K_8$-based predefined types. However, the group $G$ defined by this
    extension data is presented by
    \[
      \langle X, a \mid X^4 = a^4 = e,\ a X a^{-1} = X^{-1} \rangle,
    \]
    after eliminating the redundant generator $y = a^2$ (which lies in
    $Z(G)$ since $a^2$ commutes with $a$ and with $X$, the latter because
    $a^2 X a^{-2} = a X^{-1} a^{-1} = X$). This presentation is exactly
    that of the realising group $C_4 \rtimes_{\mathrm{inv}} C_4$ of
    $\mathtt{ext\_16\_4}$. Lemma~\ref{lem:realise-of-mulEquiv} applied to the
    induced isomorphism $G \simeq C_4 \rtimes_{\mathrm{inv}} C_4$ transfers
    $\mathtt{realise\_16\_4}$ to $G$, so this sub-sub-case also falls under
    $\mathtt{ext\_16\_4}$.
  \end{itemize}
  Hence Sub-case~2b yields the surviving new extension type
  $\mathtt{ext\_16\_2}$ (with $\tau = \mathrm{id}$), while the remaining
  $\tau \in V$ are either absorbed into $\mathtt{ext\_16\_4}$ or killed by
  minimality.

  \smallskip
  Combining Cases~1 and~2, $G$ realises one of
  \[
    \mathtt{ext\_16\_2},\ \mathtt{ext\_16\_3},\ \mathtt{ext\_16\_4},\
    \mathtt{ext\_16\_10},\ \mathtt{ext\_16\_11},\ \mathtt{ext\_16\_12},\
    \mathtt{ext\_16\_13},
  \]
  as required.
\end{proof}

\section{Main lemma (Wild's Lemma 3)}

\begin{theorem}
  \label{thm:realise-ext-type-if-not-iso-to-C2-4}
  \lean{OrderSixteen.realise_ext_type_if_not_iso_to_C2_4}
  \leanfile{M2rGroup7/SixteenClassification/Classification.lean}
  \uses{lem:realise-with-normal-C8, lem:realise-with-normal-K8}
  Let $G$ be a group of order $16$ that is not isomorphic to $(C_2)^4$. Then
  $G$ realises one of the extension types $\mathtt{ext\_16\_1}, \dots,
  \mathtt{ext\_16\_13}$.
\end{theorem}

\begin{proof}
  By the Lean lemma \texttt{exists\_normal\_C8\_or\_C4\_C2}, $G$ has either a
  normal subgroup isomorphic to $C_8$ or a normal subgroup isomorphic to
  $K_8 = C_4 \times C_2$. The two cases are handled respectively by
  Lemma~\ref{lem:realise-with-normal-C8} and
  Lemma~\ref{lem:realise-with-normal-K8}, exhausting the possibilities and
  yielding one of the thirteen extension types.
\end{proof}
